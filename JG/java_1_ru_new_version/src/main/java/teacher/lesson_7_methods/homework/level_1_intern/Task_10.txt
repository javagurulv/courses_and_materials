class FizzBuzz {

    public String detect(int number) {
        if (isDividedBy3(number) && isDividedBy5(number)) {
            return "FizzBuzz";
        } else if (isDividedBy5(number)) {
            return "Buzz";
        } else if (isDividedBy3(number)) {
            return "Fizz";
        } else {
            return "" + number;
        }
    }

    boolean isDividedBy3(int number) {
        return number % 3 == 0;
    }

    boolean isDividedBy5(int number) {
        return number % 5 == 0;
    }

}
